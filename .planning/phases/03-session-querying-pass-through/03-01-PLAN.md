use anyhow::Result;
use rusqlite::{Connection, OpenFlags, OptionalExtension as _};
use std::path::Path;

pub(crate) struct Session {
    pub(crate) id: i64,
    pub(crate) invoked_at: String,
    pub(crate) workspace_path: String,
    pub(crate) user_data_dir: Option<String>,
    pub(crate) profile: Option<String>,
    #[allow(dead_code)]
    pub(crate) local_ide_path: Option<String>,
    #[allow(dead_code)]
    pub(crate) remote_name: Option<String>,
    #[allow(dead_code)]
    pub(crate) remote_server_path: Option<String>,
    pub(crate) server_commit_hash: Option<String>,
    #[allow(dead_code)]
    pub(crate) server_bin_path: Option<String>,
    pub(crate) open_files: String,
}

pub(crate) fn open_db(path: &Path) -> Result<Connection> {
    let conn = Connection::open_with_flags(
        path,
        OpenFlags::SQLITE_OPEN_READ_WRITE
            | OpenFlags::SQLITE_OPEN_CREATE
            | OpenFlags::SQLITE_OPEN_URI
            | OpenFlags::SQLITE_OPEN_NO_MUTEX,
    )?;
    conn.execute_batch("PRAGMA busy_timeout = 5000;")?;
    Ok(conn)
}

pub(crate) fn query_latest_session(
    conn: &Connection,
    workspace: &str,
) -> Result<Option<Session>> {
    conn.query_row(
        "SELECT id, invoked_at, workspace_path, user_data_dir, profile,
                local_ide_path, remote_name, remote_server_path,
                server_commit_hash, server_bin_path, open_files
         FROM invocations
         WHERE workspace_path = ?1
         ORDER BY invoked_at DESC
         LIMIT 1",
        rusqlite::params![workspace],
        |row| {
            Ok(Session {
                id: row.get(0)?,
                invoked_at: row.get(1)?,
                workspace_path: row.get(2)?,
                user_data_dir: row.get(3)?,
                profile: row.get(4)?,
                local_ide_path: row.get(5)?,
                remote_name: row.get(6)?,
                remote_server_path: row.get(7)?,
                server_commit_hash: row.get(8)?,
                server_bin_path: row.get(9)?,
                open_files: row.get(10)?,
            })
        },
    )
    .optional()
    .map_err(anyhow::Error::from)
}

#[cfg(test)]
mod tests {

    fn make_test_db() -> Connection {

    }

    #[test]
    fn test_query_latest_session_empty_table() {
        let conn = make_test_db();
        let result = query_latest_session(&conn, "/some/path").unwrap();
        assert!(result.is_none());
    }

    #[test]
    fn test_query_latest_session_returns_most_recent() {
        let conn = make_test_db();
        conn.execute(
            "INSERT INTO invocations
             (invoked_at, workspace_path, local_ide_path, open_files)
             VALUES (?1, ?2, ?3, ?4)",
            rusqlite::params![
                "2026-04-27T10:00:00.000",
                "/home/user/project",
                "/usr/bin/code",
                "[]"
            ],
        )
        .unwrap();
        conn.execute(

        let session = query_latest_session(&conn, "/home/user/project")

        // Empty DB with no invocations table — query_latest_session returns an error

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| CLI -> SQLite DB | CLI opens user-owned DB file at a path from config or default |
| CLI -> SQL query | workspace_path string from user CLI arg is used in SQL WHERE clause |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01 | Tampering | db.rs query_latest_session | mitigate | Parameterized query with `rusqlite::params![workspace]` and `?1` placeholder -- no string interpolation in SQL (extends T-04-01 from Phase 1) |
| T-03-02 | Denial of Service | db.rs open_db | mitigate | `PRAGMA busy_timeout = 5000` prevents indefinite blocking when extension holds write lock |
| T-03-03 | Denial of Service | db.rs open_files parsing | mitigate | `serde_json::from_str().unwrap_or(json!([]))` in query.rs -- corrupt JSON does not panic the CLI |
| T-03-04 | Elevation of Privilege | config.rs db_path | accept | User-controlled path via `THIS_CODE_DB_PATH` env var opens arbitrary SQLite file; acceptable since CLI runs as the same user |
</threat_model>
