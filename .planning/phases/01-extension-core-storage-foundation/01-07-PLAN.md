Create .github/workflows/ci.yml. The .github/ directory may not exist — create it. The workflow runs on push and pull_request to main.

CI scope for Phase 1: typecheck + build + manifest checks. Full npm test (VS Code extension integration tests via @vscode/test-electron) requires Xvfb on Linux and a real GUI session; that setup is deferred to Phase 4. This CI catches the most common failures (TypeScript errors, broken builds, manifest regressions) on both platforms.

```yaml
name: CI

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

jobs:
  build:
    name: Build and typecheck (${{ matrix.os }})
    runs-on: ${{ matrix.os }}
    strategy:
      matrix:
        os: [macos-latest, ubuntu-latest]
      fail-fast: false

    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '20'
          cache: 'npm'
          cache-dependency-path: extension/package-lock.json

      - name: Install dependencies
        working-directory: extension
        run: npm ci

      - name: TypeScript typecheck
        working-directory: extension
        run: npx tsc --noEmit

      - name: Build extension bundle
        working-directory: extension
        run: npm run build

      - name: Verify manifest
        working-directory: extension
        run: |
          node -e "
            const p = require('./package.json');
            const id = p.publisher + '.' + p.name;
            if (id !== 'whardier.this-code') throw new Error('Wrong ID: ' + id);
            if (JSON.stringify(p.extensionKind) !== '[\"workspace\"]') throw new Error('extensionKind wrong');
            if (!p.activationEvents.includes('onStartupFinished')) throw new Error('activationEvents wrong');
            if (p.contributes.commands) throw new Error('commands must not exist');
            const keys = Object.keys(p.contributes.configuration.properties).sort().join(',');
            if (keys !== 'thisCode.enable,thisCode.logLevel') throw new Error('settings wrong: ' + keys);
            console.log('Manifest OK:', id);
          "

      - name: Verify dist output
        working-directory: extension
        run: test -f dist/extension.js && echo "dist/extension.js exists"

      - name: Check no save trigger
        working-directory: extension
        run: |
          if grep -r "onDidSaveTextDocument" src/; then
            echo "ERROR: onDidSaveTextDocument found in source — TRACK-05 violation"
            exit 1
          fi
          echo "TRACK-05 OK: no onDidSaveTextDocument"
```

Notes:
- `fail-fast: false` — both platforms must be independently verified; a failure on one should not skip the other
- `cache-dependency-path: extension/package-lock.json` — caches npm deps using the extension-level lockfile
- `working-directory: extension` on each step — all npm commands run from the extension subdirectory
- `npm ci` (not `npm install`) — uses lockfile for reproducible installs in CI
- Node.js 20 LTS — compatible with @vscode/sqlite3 v5.1.12-vscode prebuilts
- Full integration tests (npm run test via @vscode/test-electron) deferred to Phase 4; noted inline
- The "Check no save trigger" step enforces TRACK-05 on every CI run

```typescript
suite('PLAT-01: macOS and Linux', () => {
  test('CI workflow exists with macos-latest and ubuntu-latest matrix', () => {
    const fs = require('fs');
    const path = require('path');

    // ci.yml is at project root .github/workflows/ci.yml
    // __dirname is in extension/out/test/ or extension/src/test/ — resolve upward
    const ciPath = path.resolve(__dirname, '..', '..', '..', '..', '.github', 'workflows', 'ci.yml');
    assert.ok(fs.existsSync(ciPath), `CI workflow must exist at ${ciPath}`);

    const ciContent = fs.readFileSync(ciPath, 'utf-8');
    assert.ok(ciContent.includes('macos-latest'), 'CI must include macos-latest');
    assert.ok(ciContent.includes('ubuntu-latest'), 'CI must include ubuntu-latest');
    assert.ok(ciContent.includes('fail-fast: false'), 'CI must have fail-fast: false');
  });

  test('os.homedir() returns non-empty string on current platform', () => {
    // Validates platform path resolution works — runs on whatever OS executes the test suite
    const os = require('os');
    const homeDir = os.homedir();
    assert.ok(typeof homeDir === 'string' && homeDir.length > 0, 'os.homedir() must return a non-empty path');
    // macOS: /Users/... | Linux: /home/... or /root
    assert.ok(homeDir.startsWith('/'), 'home dir must be an absolute POSIX path on macOS/Linux');
  });
});
```
