```typescript
export async function scanExistingRemoteSessions(db: Database): Promise<void> {
  const vscodeServerBin = path.join(os.homedir(), '.vscode-server', 'bin');

  let entries: string[];
  try {
    entries = await fs.readdir(vscodeServerBin);
  } catch {
    // ~/.vscode-server/bin does not exist — local-only machine, skip silently
    return;
  }

  for (const entry of entries) {
    const entryDir = path.join(vscodeServerBin, entry);
    const sessionFile = path.join(entryDir, 'this-code-session.json');

    try {
      // Incremental: skip if already indexed by remote_server_path
      const existing = await db.get<{ id: number }>(
        'SELECT id FROM invocations WHERE remote_server_path = ? LIMIT 1',
        [entryDir]
      );
      if (existing) {
        continue;  // already in index
      }

      const raw = await fs.readFile(sessionFile, 'utf-8');
      const session = JSON.parse(raw) as {
        workspace_path?: string | null;
        user_data_dir?: string | null;
        profile?: string | null;
        local_ide_path?: string;
        remote_name?: string | null;
        remote_server_path?: string | null;
        open_files?: string[];
      };

      // Insert historical record — use entryDir as remote_server_path for dedup key
      await db.run(
        `INSERT INTO invocations
         (workspace_path, user_data_dir, profile, local_ide_path,
          remote_name, remote_server_path, open_files)
         VALUES (?, ?, ?, ?, ?, ?, ?)`,
        [
          session.workspace_path ?? null,
          session.user_data_dir ?? null,
          session.profile ?? null,
          session.local_ide_path ?? '',
          session.remote_name ?? null,
          entryDir,  // authoritative remote_server_path for dedup
          JSON.stringify(session.open_files ?? []),
        ]
      );
    } catch {
      // Missing file, malformed JSON, or DB error — skip this entry silently
      // Startup scan must never throw to caller (fire-and-forget contract)
    }
  }
}
```

Implementation notes:
- `fs.readdir(vscodeServerBin)` — reads all subdirectory names; no guarantee of 40-hex format, accept any
- Dedup check uses `remote_server_path = entryDir` — entryDir is the canonical path for the server instance
- JSON parse wrapped in same try/catch as file read — malformed files skip silently (Security: RESEARCH.md threat pattern)
- `local_ide_path` defaults to empty string `''` (NOT NULL constraint) if missing from JSON
- `open_files` defaults to empty array `[]` if missing from JSON
- The function is async and called fire-and-forget from activate() — inner errors are swallowed per contract
- Do NOT await this function in activate() (Pitfall 6 from RESEARCH.md — blocks activation)

Important: Do NOT change the writeSessionJson() function that was implemented in Plan 03. Only replace the scanExistingRemoteSessions() stub.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| ~/.vscode-server/bin/*/ filesystem → SQLite | Reads JSON files from disk written by previous VS Code Server sessions |
| session JSON content → SQLite INSERT | JSON parsed from disk — fields treated as potentially malformed |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-01 | Information Disclosure | JSON parse errors from malformed session files | mitigate | Each entry wrapped in try/catch; parse errors swallowed silently, no stack traces in OutputChannel for malformed content |
| T-05-02 | Tampering | SQLite injection via session JSON field values | mitigate | All INSERT values use parameterized queries (? placeholders); session field values never interpolated into SQL strings |
| T-05-03 | Tampering | Path traversal via entry directory names in ~/.vscode-server/bin | accept | fs.readdir returns basenames; path.join constructs full path safely; no path component from JSON is used directly in path construction (entryDir is derived from os.homedir() + hardcoded segments + readdir basename) |
| T-05-04 | Denial of Service | Large number of session files causing slow activation | accept | Startup scan is fire-and-forget (not awaited); slow scan does not affect activation latency. Tuning deferred per CONTEXT.md. |
</threat_model>
